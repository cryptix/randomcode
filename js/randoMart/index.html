func jsArr(art RandomArt) {
	fmt.Println("% put this into surface3d-d3.js")
	var arrs = make([]string, len(art))
	for j, line := range art {
		l := make([]string, len(line))
		for i, char := range line {
			l[i] = fmt.Sprintf("%d", -15*int(char))
		}
		arrs[j] = fmt.Sprintf("[%s]", strings.Join(l, ","))
	}
	fmt.Printf("{ name:'ohai',\n data: [\n%s]\n},\n", strings.Join(arrs, ",\n"))
}
